\section{Formulação e Algoritmo}

\subsection{Componentes do Modelo}
\begin{frame}[shrink=5]{Componentes do Modelo de Otimização}
  \small
  \begin{block}{Correspondência com a Instância Hídrica}
    \begin{itemize}
      \item $x_{ij}$: Vazão (fluxo) de água alocada na adutora $(i,j)$ (em l/s).
      \item $b_i$: Balanço do nó $i$. Se $b_i > 0$, é uma ETA (produção). Se $b_i < 0$, é uma RA (demanda). Se $b_i = 0$, é um nó de passagem.
      \item $u_{ij}$: Capacidade máxima física de vazão na adutora $(i,j)$ (em l/s).
      \item $c_{ij}$: Distância em linha reta em km (proxy para o custo de bombeamento).
    \end{itemize}
  \end{block}
  
  \begin{itemize}
    \item A conservação de fluxo assegura que nenhuma água ``some'' ou é adicionada em trânsito.
    \item A restrição de capacidade limita o fluxo ao máximo suportado pela tubulação física.
  \end{itemize}
\end{frame}

\subsection{Formulação Formal do MCF}
\begin{frame}{Formulação Matemática do Fluxo de Custo Mínimo}
  O problema de alocação hídrica é modelado como um problema de \textbf{Fluxo de Custo Mínimo (MCF)}:
  
  $$\min \sum_{(i,j) \in A} c_{ij} x_{ij}$$
  $$\text{sujeito a:} \quad \sum_{\{j: (i,j) \in A\}} x_{ij} - \sum_{\{j: (j,i) \in A\}} x_{ji} = b_i \quad \forall i \in V$$
  $$0 \leq x_{ij} \leq u_{ij} \quad \forall (i,j) \in A$$
\end{frame}

\subsection{Grafo Residual}
\begin{frame}{O Pulo do Gato: O Grafo Residual}
  Para resolver o MCF de forma eficiente, a ideia central é trabalhar com o \textbf{grafo residual}: uma representação dinâmica que permite ao algoritmo corrigir decisões de fluxo anteriores.

  \begin{itemize}
    \item \textbf{Arcos forward}: Representam a capacidade ainda disponível para enviar mais fluxo.
    \item \textbf{Arcos backward}: Permitem ``devolver'' fluxo já enviado, revertendo decisões subótimas.
    \item \textbf{Consequência}: O algoritmo não fica preso em soluções parciais ruins --- ele sempre pode se autocorrigir.
  \end{itemize}

  \begin{center}
    \scriptsize \textit{É essa estrutura que transforma uma heurística gulosa em um algoritmo exato.}
  \end{center}
\end{frame}

\begin{frame}{Estrutura do Grafo Residual}
  O grafo residual $G_x$ permite que o algoritmo retroceda decisões anteriores redirecionando fluxo pelas arestas reversas.
  
  \vspace{0.3cm}
  \centering
  \scalebox{0.8}{
    \begin{tikzpicture}[scale=0.9]
      % Original arc
      \node[circle, draw=unbazul, fill=unbazul!10, thick] (u) at (0, 1.5) {$i$};

\end{frame}

\subsection{Condições de Otimalidade}
\begin{frame}{Condições de Otimalidade}
  A teoria de otimização em redes fornece condições necessárias e suficientes para certificar que uma solução é ótima:

  \begin{itemize}
    \setlength{\itemsep}{10pt}
    \item \textbf{Potenciais de nó} ($\pi_i$): Variáveis duais associadas à conservação de fluxo em cada nó da rede.
    \item \textbf{Custos reduzidos} ($\bar{c}_{ij} = c_{ij} - \pi_i + \pi_j$): Ajustam os custos originais das arestas usando os potenciais, permitindo o uso de Dijkstra.
    \item \textbf{Teorema de Otimalidade} \cite{ahuja1993network}: Uma solução viável é ótima se e somente se todos os custos reduzidos no grafo residual são não-negativos ($\bar{c}_{ij} \geq 0$).
  \end{itemize}

  \begin{center}
    \scriptsize \textit{Os potenciais e custos reduzidos são a peça-chave que garante a corretude do SSP.}
  \end{center}
\end{frame}

\subsection{Pseudocódigo do SSP}
\begin{frame}[shrink=20]{Pseudocódigo do Algoritmo Successive Shortest Path}
  O SSP resolve o MCF enviando fluxo de forma incremental ao longo de caminhos mínimos.
  
  \vspace{0.1cm}
  \SetAlFnt{\footnotesize}
  \SetAlCapFnt{\footnotesize}
  \begin{algorithm}[H]
    \caption{Successive Shortest Path}
    \SetKwInOut{Input}{Entrada}\SetKwInOut{Output}{Saída}
    \Input{Grafo $G=(V,A)$, balanços $b$, capacidades $u$, custos $c$}
    \Output{Fluxo ótimo $x$}
    \SetAlgoLined
    Inicializa $x \leftarrow 0$ (fluxo nulo)\;
    Calcula potenciais iniciais $\pi$ via Bellman-Ford\;
    \While{existe nó $s$ com excesso $e(s) > 0$}{
      Escolhe nó $t$ com déficit $e(t) < 0$\;
      Encontra caminho $P$ de custo mínimo reduzido $\bar{c}$ de $s$ a $t$ em $G_x$ (Dijkstra com $\bar{c}_{ij} \geq 0$)\;
      Atualiza potenciais $\pi_i \leftarrow \pi_i - d(i)$ para todo $i \in V$ (onde $d(i)$ é a distância de $s$ a $i$)\;
      $\delta \leftarrow \min(e(s), -e(t), \min_{(i,j) \in P} r_{ij})$\;
      Aumenta $\delta$ unidades de fluxo ao longo de $P$ no grafo $x$\;
      Atualiza excessos: $e(s) \leftarrow e(s) - \delta$ e $e(t) \leftarrow e(t) + \delta$\;
    }
    \Return{$x$}\;
  \end{algorithm}
\end{frame}

\subsection{Análise de Complexidade}
\begin{frame}{Pseudocódigo Detalhado: Complexidade Nodal}
  \small
  \begin{itemize}
    \item \textbf{Estruturas de Dados}:
    \begin{itemize}
      \scriptsize
      \item Grafo residual $G_x$ como lista de adjacência dinâmica.
      \item Lista/vetor linear para busca simples de menor distância no Dijkstra.
      \item Vetores para potenciais $\pi$, excessos $e$ e fluxos ativos $x$.
    \end{itemize}
  \end{itemize}
  
  \begin{table}[h]
    \centering
    \tiny
    \begin{tabular}{llll}
      \toprule
      \textbf{Linha} & \textbf{Operação} & \textbf{Estrutura} & \textbf{Complexidade} \\
      \midrule
      1 & Inicialização de Fluxos $x=0$ & Vetor & $O(E)$ \\
      2 & Cálculo de potenciais $\pi$ iniciais & Bellman-Ford & $O(VE)$ \\
      3 & Busca por nós fontes $s$ ($e(s)>0$) & Varredura linear & $O(V)$ \\
      5 & Caminho Mínimo $P$ em $G_x$ & Dijkstra (com Lista) & $O(V^2)$ \\
      6 & Atualização de $\pi_i \leftarrow \pi_i - d(i)$ & Vetor & $O(V)$ \\
      7-9 & Envio de fluxo $\delta$ ao longo de $P$ & Busca Reversa & $O(V)$ \\
      \bottomrule
    \end{tabular}
    \caption{Resumo das estruturas e custos computacionais do SSP por iteração}
  \end{table}
\end{frame}

\begin{frame}{Análise de Complexidade}
  \begin{itemize}
    \setlength{\itemsep}{12pt}
    \item \textbf{Inicialização (Bellman-Ford)}: Executado uma única vez para calcular os potenciais iniciais $\pi$. Custo: $O(VE)$.
    \item \textbf{Custo por Iteração (Dijkstra em Lista)}: Busca por caminho mínimo com busca linear simples. Custo: $O(V^2)$ por iteração \cite{cormen2009introduction}.
    \item \textbf{Número de Iterações}: No máximo $U$ iterações, onde $U = \displaystyle\sum_{b_i > 0} b_i$ é a soma dos excessos das fontes.
    \item \textbf{Complexidade Total}: $O(VE + U \cdot V^2)$.
  \end{itemize}
\end{frame}

\subsection{Sensibilidade e Limitações}
\begin{frame}[shrink=5]{Limitações e Análise de Sensibilidade}
  Embora o modelo SSP forneça a melhor alocação sob custos lineares determinísticos, apresenta limitações estruturais:
  
  \begin{itemize}
    \item \textbf{Custo Linear vs. Custo Não-Linear}: Modelamos o custo de transporte linearmente com a distância. Em redes hidráulicas reais, a perda de carga segue modelos não-lineares (\textbf{Hazen-Williams} ou \textbf{Darcy-Weisbach}).
    \item \textbf{Pressão Nodal}: Ignoramos limites mínimos de pressão requeridos para o funcionamento dos ramais de distribuição.
    \item \textbf{Altimetria}: O relevo do Distrito Federal (topologia 3D) impõe custos adicionais de elevação hidrostática (bombeamento morro acima).
    \item \textbf{Incerteza}: As demandas são tomadas como estáticas, mas variam dinamicamente ao longo do dia.
  \end{itemize}
\end{frame}
